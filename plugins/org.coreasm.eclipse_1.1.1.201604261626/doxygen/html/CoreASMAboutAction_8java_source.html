204  @Override
205  public void run(IAction action) {
206  if (!windowOpen) {
207  display = Display.getCurrent();
208 
209  // collect content for plugin list
210  if (plugins == null) {
211  CoreASMGlobal.setRootFolder(CoreASMPlugin.getDefault().getPreferenceStore()
212  .getString(PreferenceConstants.ROOT_FOLDER));
213  CoreASMEngine engine = CoreASMEngineFactory.createCoreASMEngine();
214  engine.setClassLoader(this.getClass().getClassLoader());
215  engine.initialize();
216  engine.waitWhileBusy();
217 
218  plugins = new HashMap<String, String>();
219  if (engine.getEngineMode() != EngineMode.emError) {
220 
221  Map<String, String> pluginList = new HashMap<String, String>();
222  Map<String, VersionInfo> list = engine.getPluginsVersionInfo();
223  list.put("CoreASM Engine", engine.getVersionInfo());
224  for (Entry<String, VersionInfo> plugin : list.entrySet()) {
225  pluginList.put(plugin.getKey(), plugin.getValue().toString());
226  }
227  plugins = new TreeMap<String, String>(pluginList);
228  }
229  engine.terminate();
230  }
231  shell = createShell(display);
232  shell.open();
233  shell.layout();
234  windowOpen = true;
235  while (!shell.isDisposed()) {
236  if (!display.readAndDispatch())
237  display.sleep();
238  }
239  windowOpen = false;
240  } else {
241  shell.setActive();
242  }
243  }
