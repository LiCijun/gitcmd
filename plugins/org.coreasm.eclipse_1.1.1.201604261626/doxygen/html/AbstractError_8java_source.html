212  private static AbstractError decode(String encoded) throws Exception
213  {
214  try {
215  Map<String, String> attributes = new HashMap<String,String>();
216  String[] pairs = encoded.split(SEPERATOR_ATTR);
217  for (String pair: pairs) {
218  String[] kv = pair.split(SEPERATOR_VAL);
219  attributes.put(kv[0], kv[1]);
220  }
221 
222  ErrorType type = ErrorType.valueOf(attributes.get(AbstractError.TYPE));
223  AbstractError error;
224  switch (type) {
225  case SIMPLE:
226  error = new SimpleError(attributes);
227  break;
228  case SYNTAX_ERROR:
229  error = new SyntaxError(attributes);
230  break;
231  case UNDEFINED:
232  error = new UndefinedError(attributes);
233  break;
234  case COREASM_ERROR:
235  error = new CoreASMEclipseError(attributes);
236  break;
237  default:
238  error = null;
239  }
240 
241  return error;
242 
243  }catch(Exception e) {
244  //@warning there maybe unhandled error types which will cause an exception during decoding
245  throw new Exception("Exception in AbstractError.decode("+encoded+") with message\n"+e.getMessage());
246  }
