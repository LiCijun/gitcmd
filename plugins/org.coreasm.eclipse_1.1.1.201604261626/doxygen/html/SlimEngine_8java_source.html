99  continue;
100  }

104  // create new instance for each plugin which is bound to this engine
105  for (Plugin p: tmpPlugins) {
106  Plugin p2 = null;

118  if (extensionPointPlugin.getSourceModes() != null && extensionPointPlugin.getSourceModes().containsKey(EngineMode.emParsingSpec))

211  private static void createFullEngine()
212  {
213  fullEngine = (ControlAPI) CoreASMEngineFactory.createCoreASMEngine();
214  Set<Plugin> plugins = new HashSet<Plugin>();
215 
216  // get the private allPlugins map from the engine
217  try {
218  Field f = Engine.class.getDeclaredField("allPlugins");
219  f.setAccessible(true);
220  Object o = f.get(fullEngine);
221  Map<String,Plugin> m = (Map<String, Plugin>) o;
222  plugins = new HashSet<Plugin>(m.values());
223  } catch (Exception e) {
224  e.printStackTrace();
225  }
226 
227  // build a specification with an use clause for each plugin
228  StringBuilder strSpec = new StringBuilder();
229  strSpec.append("CoreASM InitEngine\n");
230  for (Plugin p: plugins)
231  strSpec.append("use ").append(p.getName()).append("\n");
232  strSpec.append("init main\n");
233  strSpec.append("rule main = skip");
234 
235  // load all plugins
236  getSpec(strSpec.toString(), true, (Engine) fullEngine);
237  fullEngine.terminate();
238  }
