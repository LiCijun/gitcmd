416  public void updateState(ASTNode pos) {
417  ASMStorage state = null;
418  Stack<CallStackElement> callStack = capi.getInterpreter().getInterpreterInstance().getCurrentCallStack();
419  Map<String, Element> envVars = capi.getInterpreter().getInterpreterInstance().getEnvVars();
420  if (stepSucceeded) {
421  if (!states.isEmpty() && capi.getStepCount() == states.peek().getStep())
422  return;
423  while (!states.isEmpty() && states.peek().getStep() < 0)
424  states.pop();
425  state = new ASMStorage(wapi, capi.getStorage(), capi.getStepCount(), capi.getLastSelectedAgents(), envVars, updates, capi.getAgentSet(), callStack, sourceName, lineNumber);
426  }
427  else {
428  if (!states.isEmpty() && states.peek().getStep() < 0) {
429  if (updates.isEmpty() && states.peek().getUpdates().isEmpty() && envVars.equals(states.peek().getEnvVars()))
430  state = states.pop();
431  else
432  state = new ASMStorage(states.peek());
433  }
434  Set<Element> agent = new HashSet<Element>();
435  agent.add(currentAgent);
436  if (state == null)
437  state = new ASMStorage(wapi, capi.getStorage(), -capi.getStepCount() - 1, agent, envVars, updates, capi.getAgentSet(), callStack, sourceName, lineNumber);
438  state.updateState(pos, agent, envVars, updates, callStack, sourceName, lineNumber);
439  }
440  if (ruleArgs != null) {
441  final ASMStorage[] statePointer = new ASMStorage[] { state };
442  for (Entry<ASTNode, String> arg : ruleArgs.entrySet()) {
443  final ASTNode node = (ASTNode)arg.getKey().cloneTree();
444  Display.getDefault().syncExec(new Runnable() {
445 
446  @Override
447  public void run() {
448  try {
449  wapi.evaluateExpression(node, currentAgent, statePointer[0]);
450  } catch (InterpreterException e) {
451  }
452  }
453  });
454  if (node.isEvaluated())
455  envVars.put(arg.getValue(), node.getValue());
456  }
457  }
458  states.add(state);
459  updates = new HashSet<ASMUpdate>();
460  }
