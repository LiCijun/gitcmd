147  public static IFile createFile(String filename, IProject project)
148  {
149  IFile newfile = project.getFile(filename);
150  String text = "// CoreASM specification";
151  InputStream is = null;
152  try {
153  is = new ByteArrayInputStream(text.getBytes("UTF-8"));
154  } catch (UnsupportedEncodingException e) {
155  e.printStackTrace();
156  }
157  try {
158  // create folders if needed
159  boolean folderWasCreated = false;
160  IPath path = newfile.getLocation().removeLastSegments(1);
161  for (int i=path.segmentCount(); i>=0; i--) {
162  IPath fPath = path.removeLastSegments(i);
163 
164  java.io.File jFile = new File(fPath.toString());
165  if (jFile.exists() && jFile.isDirectory()==false)
166  throw new RuntimeException("cannot create directory");
167  if (!jFile.exists())
168  folderWasCreated = jFile.mkdir();
169  }
170 
171  if (folderWasCreated) {
172  getActiveProject().refreshLocal(IResource.DEPTH_INFINITE, null);
173  Logger.log(Logger.INFORMATION, Logger.ui, "created directory: " + newfile.getProjectRelativePath().removeLastSegments(1).toString());
174  }
175 
176  newfile.create(is, false, null);
177  } catch (CoreException e) {
178  e.printStackTrace();
179  }
180  return newfile;
181  }
