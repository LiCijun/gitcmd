31  if (root == null)
32  return;
33  IDocumentProvider documentProvider = editor.getDocumentProvider();
34  ASMDocument document = (ASMDocument)documentProvider.getDocument(editor.getEditorInput());
35  IAnnotationModel annotationModel = documentProvider.getAnnotationModel(editor.getEditorInput());
36  ASTNode node = document.getIDnodeAt(selection.getOffset());
37  Map<Annotation, Position> annotationMap = new HashMap<Annotation, Position>();
38  if (node != null) {
39  Stack<ASTNode> fringe = new Stack<ASTNode>();
40 
41  fringe.add(root);
42  while (!fringe.isEmpty()) {
43  ASTNode n = fringe.pop();
44  if (ASTNode.ID_CLASS.equals(n.getGrammarClass()) && node.getToken().equals(n.getToken())) {
45  String type = ANNOTATION_OCCURRENCE;
46  if (n.getParent() != null && n.getParent().getParent() != null) {
47  ASTNode grandParent = n.getParent().getParent();
48  if ((grandParent instanceof UpdateRuleNode || grandParent instanceof ReturnResultNode) && grandParent.getFirst() == n.getParent())
49  type = ANNOTATION_OCCURRENCE_WRITE;
50  }
51  Annotation annotation = new Annotation(type, false, (ANNOTATION_OCCURRENCE_WRITE.equals(type) ? "Write Occurrence" : "Occurrence") + " of " + n.getToken());
52  annotationMap.put(annotation, new Position(document.getNodePosition(n), n.getToken().length()));
53  }
54  for (ASTNode child : n.getAbstractChildNodes())
55  fringe.add(fringe.size(), child);
56  }
57  if (annotationModel instanceof IAnnotationModelExtension)
58  ((IAnnotationModelExtension)annotationModel).replaceAnnotations(annotations, annotationMap);
59  else {
60  for (Entry<Annotation, Position> annotation : annotationMap.entrySet())
61  annotationModel.addAnnotation(annotation.getKey(), annotation.getValue());
62  }
63  annotations = annotationMap.keySet().toArray(new Annotation[annotationMap.keySet().size()]);
64  }
65  }
