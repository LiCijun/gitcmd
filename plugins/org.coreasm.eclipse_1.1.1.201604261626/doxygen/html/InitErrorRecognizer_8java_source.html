34  @Override
35  public void checkForErrors(ASMDocument document, List<AbstractError> errors)
36  {
37  ASTNode root = (ASTNode) document.getRootnode();
38 
39  // get a list with all initialization nodes
40  List<ASTNode> initList = AstTools.findChildNodes(root, AstTools.GRAMMAR_INIT);
41 
42  // check if there is at least one init node
43  if (initList.size() == 0 && !((ASMDocument) document).isIncludedSpecification()) {
44  String id = AstTools.findId(root);
45  // Mark "CoreASM ID" for the error
46  int length = 8 + id.length();
47  AbstractError error = new SimpleError("No \"init\" statement", "specification \"" + id + "\" contains no \"init\" statement", root, document, length, CLASSNAME, NO_INIT);
48  errors.add(error);
49  }
50  // check if there are more than one init nodes
51  else if (initList.size() > 1) {
52  for (ASTNode inode: initList) {
53  String id = AstTools.findId(inode);
54  // Mark the whole init statement ("init" and ID)
55  int length = 5 + id.length();
56  AbstractError error = new SimpleError("Multiple \"init\" statements", "specification contains multiple \"init\" statements", inode, document, length, CLASSNAME, MULTI_INITS);
57  errors.add(error);
58  }
59  }
60 
61  // check if the rule for each init node is existing
62  for (ASTNode inode: initList) {
63  ASTNode rnode = getInitRuleDefinition(root, inode);
64  if (rnode == null) {
65  Node idNode = AstTools.findIdNode(inode);
66  // Mark only the ID
67  int length = idNode.getToken().length();
68  String name = AstTools.findId(inode);
69  String msg = "There is no rule \"" + name + "\"";
70  AbstractError error = new SimpleError("Undeclared initalization rule", msg, idNode, document, length, CLASSNAME, UNKN_INIT);
71  errors.add(error);
72  }
73  }

238  public void fix(AbstractError error, String choice)
239  {
240  if (error instanceof SimpleError && error.getDocument() instanceof ASMDocument)
241  {
242  ASMDocument doc = (ASMDocument) error.getDocument();
243  Node rootnode = doc.getRootnode();
244  List<ASTNode> initnodes = AstTools.findChildNodes((ASTNode) rootnode, AstTools.GRAMMAR_INIT);
245  // walk through list & delete inits backwards, so the indexes
246  // of the other inits don't get messed up because of the deletions.
247  for (int i=initnodes.size()-1; i>=0; i--) {
248  ASTNode initnode = initnodes.get(i);
249  String id = AstTools.findId(initnode);
250  if (id.equals(choice)) // skip the chosen init
251  continue;
252  int begin = doc.getNodePosition(initnode);
253  int len = 5 + id.length();
254  // check if the whole line can be deleted (when there are only whitespaces left)
255  try {
256  int lineNr = doc.getLineOfOffset(begin);
257  int lineBegin = doc.getLineOffset(lineNr);
258  int lineEnd = doc.getLineOffset(lineNr+1);
259  String sBefore = doc.get().substring(lineBegin,begin);
260  String sEnd = doc.get().substring(begin+len,lineEnd);
261  if (sBefore.matches("\\s*") && sEnd.matches("\\s*")) {
262  begin = lineBegin;
263  len = lineEnd-lineBegin;
264  }
265  } catch (BadLocationException e1) {
266  e1.printStackTrace();
267  }
268  try {
269  doc.replace(begin, len, "");
270  } catch (BadLocationException e) {
271  e.printStackTrace();
272  }
273  }
274  }
275  }
