43  @Override
44  public List<AbstractWarning> checkForWarnings(ASMDocument document) {
45  List<AbstractWarning> warnings = new LinkedList<AbstractWarning>();
46  HashMap<String, Declaration> declarations = new HashMap<String, Declaration>();
47  Stack<ASTNode> fringe = new Stack<ASTNode>();
48 
49  for (Declaration declaration : ASMDeclarationWatcher.getDeclarations(parentEditor.getInputFile(), true)) {
50  if (!(declaration instanceof RuleDeclaration))
51  declarations.put(declaration.getName(), declaration);
52  }
53 
54  for (ASTNode declarationNode = ((ASTNode)document.getRootnode()).getFirst(); declarationNode != null; declarationNode = declarationNode.getNext()) {
55  if (ASTNode.DECLARATION_CLASS.equals(declarationNode.getGrammarClass())) {
56  if (Kernel.GR_RULEDECLARATION.equals(declarationNode.getGrammarRule())
57  || "Signature".equals(declarationNode.getGrammarRule()) && declarationNode.getFirst() instanceof DerivedFunctionNode) {
58  fringe.add(declarationNode);
59  while (!fringe.isEmpty()) {
60  ASTNode node = fringe.pop();
61  if (ASTNode.FUNCTION_RULE_CLASS.equals(node.getGrammarClass()) && node instanceof FunctionRuleTermNode) {
62  if (node.getParent() instanceof MacroCallRuleNode) // Wrong number of arguments for rules cause an error
63  break;
64  FunctionRuleTermNode frNode = (FunctionRuleTermNode)node;
65  if (frNode.hasName()) {
66  Declaration declaration = declarations.get(frNode.getName());
67  if (declaration instanceof FunctionDeclaration) {
68  FunctionDeclaration functionDeclaration = (FunctionDeclaration)declaration;
69  if (!frNode.hasArguments()) {
70  if (functionDeclaration.getDomain().size() > 0 && !isEnvironmentVariable(frNode) && !isLocalFunction(frNode))
71  warnings.add(new NumberOfArgumentsWarning(frNode.getName(), 0, functionDeclaration.getDomain(), frNode, document));
72  }
73  else if (functionDeclaration.getDomain().size() != frNode.getArguments().size())
74  warnings.add(new NumberOfArgumentsWarning(frNode.getName(), frNode.getArguments().size(), functionDeclaration.getDomain(), frNode, document));
75  }
76  else if (declaration instanceof UniverseDeclaration || declaration instanceof EnumerationDeclaration) {
77  if (frNode.hasArguments()) {
78  if (frNode.getArguments().size() > 1)
79  warnings.add(new NumberOfArgumentsWarning(frNode.getName(), frNode.getArguments().size(), Arrays.asList(new String[] { "ELEMENT" }), frNode, document));
80  }
81  }
82  else if (declaration instanceof DerivedFunctionDeclaration) {
83  DerivedFunctionDeclaration functionDeclaration = (DerivedFunctionDeclaration)declaration;
84  if (!frNode.hasArguments()) {
85  if (functionDeclaration.getParams().size() > 0 && !isEnvironmentVariable(frNode) && !isLocalFunction(frNode))
86  warnings.add(new NumberOfArgumentsWarning(frNode.getName(), 0, functionDeclaration.getParams(), frNode, document));
87  }
88  else if (functionDeclaration.getParams().size() != frNode.getArguments().size())
89  warnings.add(new NumberOfArgumentsWarning(frNode.getName(), frNode.getArguments().size(), functionDeclaration.getParams(), frNode, document));
90  }
91  }
92  }
93  fringe.addAll(node.getAbstractChildNodes());
94  }
95  }
96  }
97  }
98 
99  return warnings;
100  }
