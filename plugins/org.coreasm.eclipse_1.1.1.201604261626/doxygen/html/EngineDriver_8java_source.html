247  public void run()
248  {
249  if (this == runningInstance)
250  updateStatus(EngineDriverStatus.running);
251 
252  int step=0;
253  Exception exception = null;
254 
255  engine.addObserver(this); // TODO this too prevents more than a single syntaxInstance being run at the same time...
256  Set<Update> updates,prevupdates=null;
257 
258  try {
259 
260  setInputOutputPhase1();
261 
262  if (engine.getEngineMode() == EngineMode.emError) {
263  engine.recover();
264  engine.waitWhileBusy();
265  }
266  engine.loadSpecification(abspathname);
267  engine.waitWhileBusy();
268  if (engine.getEngineMode()!=EngineMode.emIdle) {
269  handleError();
270  return;
271  }
272 
273  if (shouldStop)
274  throw new EngineDriverException();
275 
276  setInputOutputPhase2();
277 
278  preExecutionCallback();
279 
280  clearEclipseRuntimeErrors();
281 
282  while (engine.getEngineMode()==EngineMode.emIdle) {
283  if (shouldPause) {
284  if (this == runningInstance)
285  updateStatus(EngineDriverStatus.paused);
286 
287  stderr.println("[!] Run is paused by user. Click on resume to continue...");
288 
289  while (shouldPause && !shouldStop)
290  Thread.sleep(100);
291 
292  if (!shouldStop)
293  stderr.println("[!] Resuming.");
294 
295  if (this == runningInstance && !shouldStop)
296  updateStatus(EngineDriverStatus.running);
297  }
298 
299  if (shouldStop) {
300  throw new EngineDriverException();
301  }
302 
303  engine.step(); step++;
304 
305  while (!shouldStop && engine.isBusy())
306  Thread.sleep(50);
307 
308  if (shouldStop) {
309  // give some time to the engine to finish
310  if (engine.isBusy())
311  Thread.sleep(200);
312 
313  throw new EngineDriverException();
314  }
315 
316  updates = engine.getUpdateSet(0);
317  if (markSteps)
318  stddump.println("#--- end of step " + step);
319  if (dumpUpdates)
320  stddump.println("Updates at step "+step+": "+updates);
321  if (dumpState)
322  stddump.println("State at step "+step+":\n"+engine.getState());
323  if (printAgents)
324  stddump.println("Last selected agents: " + engine.getLastSelectedAgents());
325  if (terminated(step,updates,prevupdates))
326  break;
327  prevupdates=updates;
328 
329  }
330  if (engine.getEngineMode()!=EngineMode.emIdle)
331  handleError();
332  } catch (Exception e) {
333  exception = e;
334  } finally {
335  engine.removeObserver(this);
336  if (exception != null)
337  if (exception instanceof EngineDriverException)
338  stderr.println("[!] Run is terminated by user.");
339  else {
340  stderr.println("[!] Run is terminated with exception " + exception);
341  }
342 
343  if (dumpFinal && step > 0) {
344  stddump.println("--------------------FINISHED---------------------");
345  stddump.println("Final engine mode was "+engine.getEngineMode());
346  if (lastError != null)
347  stddump.println("Last error was "+lastError);
348  if (stepFailedMsg != null)
349  stddump.println("Step failed reason was "+stepFailedMsg);
350  stddump.println("Final state was:\n"+engine.getState());
351  //stddump.println("Output history:\n"+getOutputString());
352 
353  // Repeating
354  if (exception != null)
355  if (exception instanceof EngineDriverException)
356  stderr.println("[!] Run is terminated by user.");
357  else
358  stderr.println("[!] Run is terminated with exception " + exception);
359  }
360  System.setErr(systemErr);
361  engine.terminate();
362 
363  if (this == runningInstance)
364  updateStatus(EngineDriverStatus.stopped);
365 
366  runningInstance.engine.hardInterrupt();
367 
368  runningInstance = null;
369 
370  postExecutionCallback();
371  }
372  }

374  private boolean terminated(int step, Set<Update> updates, Set<Update> prevupdates) {
375  if (stopOnEmptyUpdates && updates.isEmpty())
376  return true;
377  if (stopOnStableUpdates && updates.equals(prevupdates))
378  return true;
379  if (stopOnEmptyActiveAgents && engine.getAgentSet().size() < 1)
380  return true;
381  if (stopOnFailedUpdates && updateFailed)
382  return true;
383  if (stopOnError && lastError!=null)
384  return true;
385  if (stopOnStepsLimit && step>stepsLimit)
386  return true;
387  return false;
388  }

553  public void update(EngineEvent event) {
554 
555  // Looking for StepFailed
556  if (event instanceof StepFailedEvent) {
557  StepFailedEvent sEvent = (StepFailedEvent)event;
558  synchronized (this) {
559  updateFailed = true;
560  stepFailedMsg = sEvent.reason;
561  ControlAPI capi = (ControlAPI)engine;
562  for (Update update : capi.getStorage().getLastInconsistentUpdate()) {
563  for (ScannerInfo scannerInfo : update.sources)
564  showErrorInEclipse(new CoreASMError("Inconsistent Update: " + update, scannerInfo.getPos(capi.getParser().getPositionMap())));
565  }
566  }
567  }
568 
569  // Looking for errors
570  else if (event instanceof EngineErrorEvent) {
571  synchronized (this) {
572  lastError = ((EngineErrorEvent)event).getError();
573  }
574  showErrorInEclipse(lastError);
575  }
576  else if (event instanceof EngineWarningEvent)
577  showWarningInEclipse(((EngineWarningEvent)event).getWarning());
578  }
