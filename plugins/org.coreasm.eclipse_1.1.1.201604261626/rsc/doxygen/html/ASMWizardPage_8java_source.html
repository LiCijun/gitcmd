146  private void dialogChanged() {
147  IResource container = ResourcesPlugin.getWorkspace().getRoot()
148  .findMember(new Path(getContainerName()));
149  String fileName = getFileName();
150 
151  if (getContainerName().length() == 0) {
152  updateStatus("File container must be specified");
153  return;
154  }
155  if (container == null
156  || (container.getType() & (IResource.PROJECT | IResource.FOLDER)) == 0) {
157  updateStatus("File container must exist");
158  return;
159  }
160  if (!container.isAccessible()) {
161  updateStatus("Project must be writable");
162  return;
163  }
164  if (fileName.length() == 0) {
165  updateStatus("File name must be specified");
166  return;
167  }
168  if (fileName.replace('\\', '/').indexOf('/', 1) > 0) {
169  updateStatus("File name must be valid");
170  return;
171  }
172  int dotLoc = fileName.lastIndexOf('.');
173  if (dotLoc != -1) {
174  String ext = fileName.substring(dotLoc + 1);
175  if (ext.equalsIgnoreCase(CoreASMPlugin.COREASM_FILE_EXT_1) == false
176  && ext.equalsIgnoreCase(CoreASMPlugin.COREASM_FILE_EXT_2) == false) {
177  updateStatus("File extension must be \""
178  + CoreASMPlugin.COREASM_FILE_EXT_1 + "\" or \""
179  + CoreASMPlugin.COREASM_FILE_EXT_2 + "\".");
180  return;
181  }
182  }
183  updateStatus(null);
184  }
