117  else
118  proposals.add(new CompletionProposal(declaration.getName(), start, end - start, declaration.getName().length(), IconManager.getIcon("/icons/editor/bullet.gif"), "Replace with '" + declaration.getName() + "'", null, null));
119  }
120  else if (declaration instanceof FunctionDeclaration) {
121  FunctionDeclaration functionDeclaration = (FunctionDeclaration)declaration;
122  if (functionDeclaration.getDomain().size() > 0)
123  proposals.add(new CompletionProposal(declaration.getName() + "()", start, end - start, declaration.getName().length() + 1, IconManager.getIcon("/icons/editor/bullet.gif"), "Replace with '" + declaration.getName() + "()'", null, null));
124  else

128  proposals.add(new CompletionProposal(declaration.getName(), start, end - start, declaration.getName().length(), IconManager.getIcon("/icons/editor/bullet.gif"), "Replace with '" + declaration.getName() + "'", null, null));
129  }

243  private static boolean isSimilar(String string, String anotherString) {
244  String shortString = string.toLowerCase();
245  String longString = anotherString.toLowerCase();
246 
247  if (shortString.length() > longString.length()) {
248  String tmp = longString;
249  longString = shortString;
250  shortString = tmp;
251  }
252 
253  int lengthShortString = shortString.length();
254  int lengthLongString = longString.length();
255  int lengthDiff = lengthLongString - lengthShortString;
256  int min = lengthLongString;
257 
258  for (int i = 0; i <= lengthDiff; i++) {
259  int sum = lengthDiff;
260  for (int j = 0; j < lengthShortString; j++) {
261  if (longString.charAt(i + j) != shortString.charAt(j)) {
262  sum += 2;
263  if (sum >= min)
264  break;
265  }
266  }
267  if (sum < min) {
268  min = sum;
269  if (min < lengthShortString)
270  return true;
271  }
272  }
273  int sum = lengthDiff;
274  for (int i = 0; i < lengthShortString / 2; i++) {
275  if (longString.charAt(i) != shortString.charAt(i))
276  sum += 2;
277  if (longString.charAt(lengthLongString - i - 1) != shortString.charAt(lengthShortString - i - 1))
278  sum += 2;
279  if (sum >= min)
280  break;
281  }
282  if (sum < min) {
283  min = sum;
284  if (min < lengthShortString)
285  return true;
286  }
287 
288  return false;
289  }
