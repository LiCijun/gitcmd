155  @Override
156  public void update(Observable o, Object arg)
157  {
158  // check if correctly called by the right parser
159  if (o != asmEditor.getParser() ||
160  ! (arg instanceof ParsingResult))
161  return;
162  ParsingResult result = (ParsingResult) arg;
163  List<AbstractError> errors = new LinkedList<AbstractError>();
164 
165  // clear old markers
166  asmEditor.removeMarkers(IMarker.PROBLEM);
167 
168  // always run TextErrorRecognizers
169  errors.addAll(checkAllTextErrorRecognizers(result.document));
170 
171  // run TreeErrorRecognizers only if there was no syntax error
172  if (result.wasSuccessful == true)
173  errors.addAll(checkAllTreeErrorRecognizers(result.document));
174 
175  // create markers for all errors
176  for (AbstractError error: errors) {
177  if (error instanceof SimpleError)
178  asmEditor.createSimpleMark((SimpleError) error, IMarker.SEVERITY_ERROR);
179  else
180  asmEditor.createErrorMark(error);
181  }
182 
183  if (result.wasSuccessful) {
184  for (AbstractWarning warning : checkAllWarnings(result.document))
185  asmEditor.createWarningMark(warning);
186  }
187 
188  // if there was a syntax error: create error object
189  if (result.exception != null) {
190  ParserException pe = result.exception;
191  ParseErrorDetails perr = pe.getErrorDetails();
192 
193  if (perr != null) {
194  // SYNTAX ERROR
195  int line = asmEditor.getSpec().getLine(pe.getLocation().line).line;
196  int col = pe.getLocation().column;
197  int index = 0;
198  try {
199  if (line > 0)
200  index = result.document.getLineOffset(line-1) + col-1;
201  } catch (BadLocationException e1) {
202  e1.printStackTrace();
203  }
204  String message = pe.getMessage();
205  int beginIndex = message.indexOf(':');
206  if (beginIndex > 0)
207  message = "Syntax Error: " + message.substring(beginIndex + 1).trim();
208  String encountered = perr.getEncountered().trim();
209  int length = getErrorLength(encountered, result.document.get(), index);
210  // build expected string
211  List<String> lstExpected = perr.getExpected();
212  deleteDuplicatesAndSortList(lstExpected);
213  // create error object
214  SyntaxError serror = new SyntaxError(message, line, col, index, length, lstExpected, encountered);
215  asmEditor.createSyntaxMark(serror, IMarker.SEVERITY_ERROR);
216  }
217  else
218  { // OTHER ERROR
219  String message = pe.getMessage();
220  int line = pe.getLocation().line;
221  int col = pe.getLocation().column;
222  // create error object
223  UndefinedError error = new UndefinedError(message, line, col);
224  asmEditor.createUndefinedMark(error, IMarker.SEVERITY_ERROR);
225  }
226 
227  }

237  private int getErrorLength(String token, String strDoc, int index)
238  {
239  if (token.equals("EOF"))

241  if (strDoc.charAt(index)=='"' && strDoc.startsWith(token, index+1))
242  return token.length()+2;
243  return token.length();
244  }
