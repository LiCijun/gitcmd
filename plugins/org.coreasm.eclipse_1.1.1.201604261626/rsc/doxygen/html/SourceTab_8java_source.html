163  public boolean isValid(ILaunchConfiguration launchConfig) {
164  setErrorMessage(null);
165  if (comp.getProject().getText().isEmpty()) {
166  setErrorMessage("You must choose a project.");
167  return false;
168  }
169  if (comp.getSpec().getText().isEmpty()) {
170  setErrorMessage("You must choose a specification (."
171  + CoreASMPlugin.COREASM_FILE_EXT_1 + " or ."
172  + CoreASMPlugin.COREASM_FILE_EXT_2 + ") file.");
173  return false;
174  }
175  IResource resource = ResourcesPlugin.getWorkspace().getRoot().findMember(comp.getProject().getText());
176  if (!(resource instanceof IProject)) {
177  setErrorMessage("'" + comp.getProject().getText() + "' is no project");
178  return false;
179  }
180  IProject project = (IProject)resource;
181  resource = project.findMember(comp.getSpec().getText());
182  if (resource == null) {
183  setErrorMessage("Cannot find specification '" + comp.getSpec().getText() + "' inside project '" + project.getName() + "'");
184  return false;
185  }
186  return super.isValid(launchConfig);
187  }
