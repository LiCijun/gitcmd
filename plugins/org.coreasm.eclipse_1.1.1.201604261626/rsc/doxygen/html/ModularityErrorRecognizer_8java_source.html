170  private List<IncludeStatement> getIncludes(IDocument document)
171  {
172  List<IncludeStatement> includes = new LinkedList<IncludeStatement>();
173 
174  Pattern pattern = Pattern.compile(PATTERN_RAW);
175  Matcher matcher = pattern.matcher(document.get());
176 
177  while (matcher.find() == true) {
178  String source = matcher.group(0);
179  String name = matcher.group(1);
180  int pos = matcher.start(0);
181  int length = source.length();
182  String partitiontype = "";

186  } catch (BadLocationException e) {
187  e.printStackTrace();
188  }
189  IncludeStatement is = new IncludeStatement(pos, length, source, name, partitiontype);
190  includes.add(is);
191  }

200  private void checkLineWithError(IncludeStatement include, String line, List<AbstractError> errors)
201  {
202  int offset = line.indexOf("include");
203 
204  // Check if there is something except whitespaces before the include
205  String before = line.substring(0, offset);
206  if ( ! before.matches(PATTERN_BEFORE)) {
207  String title = "Code before \"include\"";
208  String message = "There is code or a multiline comment preciding an include statement within the same line.\nThe CoreASM engine will ignore this include statement";
209  // Mark everything before the include statement, except leading and trailing whitespaces.
210  int spacesbefore = 0;
211  while (Character.isWhitespace(before.charAt(spacesbefore)))
212  spacesbefore++;
213  int spacesafter = 0;
214  while (Character.isWhitespace(before.charAt(before.length()-1-spacesafter)))
215  spacesafter++;
216  int pos = include.position - (before.length() - spacesbefore);
217  int length = before.length() - spacesbefore - spacesafter;
218  AbstractError error = new SimpleError(title, message, pos, length, CLASSNAME, CODE_BEFORE);
219  errors.add(error);
220  }
221 
222  // Check if there is something except whitespaces after the include
223  String after = line.substring(offset+include.length);
224  if ( ! after.matches(PATTERN_AFTER)) {
225  String title = "Code or comment after \"include\"";
226  String message = "There is code or a comment following an include statement within the same line.\nThis causes the CoreASM engine to throw an error.";
227  // Mark everything after the include statement, except leading and trailing whitespaces.
228  int spacesbefore = 0;
229  while (Character.isWhitespace(after.charAt(spacesbefore)))
230  spacesbefore++;
231  int spacesafter = 0;
232  while (Character.isWhitespace(after.charAt(after.length()-1-spacesafter)))
233  spacesafter++;
234  int pos = include.position + include.length + spacesbefore;
235  int length = after.length() - spacesbefore - spacesafter;
236  AbstractError error = new SimpleError(title, message, pos, length, CLASSNAME, CODE_AFTER);
237  errors.add(error);
238  }
