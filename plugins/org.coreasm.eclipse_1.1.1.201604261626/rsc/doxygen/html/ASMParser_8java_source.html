138  private void parseDocument()
139  {
140  long tStart = System.currentTimeMillis();
141  String filename = parentEditor.getInputFile().getProjectRelativePath().toString();
142  //System.out.print("=====================================\nparsing... (" + filename + ") ");
143 
144  StringBuilder logmsg = new StringBuilder(100);
145  logmsg.append("parsed: ").append(filename).append(" ... ");
146 
147  ASMDocument doc = (ASMDocument) documentProvider.getDocument(parentEditor.getInput());
148  if (doc == null || doc.get().isEmpty())
149  return;
150  ParsingResult result = null;
151 
152  // forget if the document is a module or not, so it will be rechecked
153  // on the next call of isIncludedSpecification()
154  doc.resetInclusionState();
155 
156  initParser(); // create a new parser if necessary
157 
158  positionMap = null;
159  try {
160  Specification spec = new Specification(slimengine, new StringReader(doc.get()), parentEditor.getInputFile().getLocation().toFile().getAbsolutePath());
161  spec.setPluginNames(getUsedPlugins());
162  ((SlimEngine)slimengine).setSpec(spec);
163  } catch (IOException e) {
164  }
165  ((SlimEngine)slimengine).notifyEngineParsing();
166 
167  try {
168  ParserTools parserTools = ParserTools.getInstance(slimengine);
169  Parser<Node> parser;
170  if (doc.isIncludedSpecification())
171  // specification is a module -> run module parser
172  parser = getModuleParser().from(parserTools.getTokenizer(), parserTools.getIgnored());
173  else
174  // specification is not a module -> run root parser
175  parser = rootParser.from(parserTools.getTokenizer(), parserTools.getIgnored());
176 
177  rootnode = parser.parse(getSpec().getText());
178  doc.setRootnode(rootnode);
179  doc.setControlAPI(slimengine);
180 
181  //System.out.println("correct");
182  logmsg.append("correct ");
183  result = new ParsingResult(true, doc, null); // result for the observers
184 
185  } catch (ParserException pe) {
186  // if we reach this catch block there was an exception during parsing.
187 
188  ParseErrorDetails perr = pe.getErrorDetails();
189 
190  rootnode = null;
191  doc.setRootnode(null);
192 
193  if (perr != null) {
194  // SYNTAX ERROR (there is a ParseErrorDetails object)
195  //System.out.println("SYNTAX ERROR");
196  logmsg.append("SYNTAX ERROR ");
197  }
198  else {
199  // UNKNOWN ERROR (there is no ParseErrorDetails object)
200  // an unknown error should not appear and usually is an indication
201  // for an error in the implementation of the parser.
202  // The stack trace is printed for information about the error.
203  //System.out.println("unknown error");
204  logmsg.append("UNKNOWN ERROR ");
205  pe.printStackTrace();
206  }
207 
208  result = new ParsingResult(false, doc, pe); // result for the observers
209  }
210  catch (Exception e) {
211  e.printStackTrace();
212  }
213 
214  parentEditor.createPluginMark(uses);
215 
216  // notify observers
217  if (result.wasSuccessful)
218  ((SlimEngine)slimengine).notifyEngineParsingFinished();
219  setChanged();
220  notifyObservers(result);
221 
222  long tStop = System.currentTimeMillis();
223  long t = tStop - tStart;
224  //System.out.println("t=" + t + "ms");
225  logmsg.append('(').append(t).append("ms)");
226  Logger.log(Logger.INFORMATION, Logger.ui, logmsg.toString());
227  }

276  private Set<String> getUsedPlugins()
277  {
278  ControlAPI engine = SlimEngine.getFullEngine();
279 
280  IDocument doc = documentProvider.getDocument(parentEditor.getInput());
281  String strDoc = doc.get();
282 
283  uses.clear();
284 
285  Set<String> pluginnames = new HashSet<String>();
286  pluginnames.add("Kernel");
287  if (((ASMDocument)doc).isIncludedSpecification()) {
288  pluginnames.add("ModularityPlugin");
289  pluginnames.add("StringPlugin"); // Modularity depends on String
290  pluginnames.add("NumberPlugin"); // String depends on Number
291  }
292 
293  // get plugin names from correct use statements
294  // find only those use statements who are in their own line.
295  Pattern p = Pattern.compile("^\\s*use\\s+(\\w+)\\s*$", Pattern.MULTILINE);
296  Matcher m = p.matcher(strDoc);
297  while (m.find() == true) {
298  try {
299  // ignore commented uses
300  if ( ! doc.getContentType(m.start()).equals(ASMEditor.PARTITION_CODE))
301  continue;
302  } catch (BadLocationException e) {
303  e.printStackTrace();
304  }
305 
306  String pluginname = m.group(1);
307  pluginnames.add(pluginname);
308  }
309 
310  // get plugin objects for the found plugin names
311  Set<Plugin> plugins = new HashSet<Plugin>();
312  for (String pluginname: pluginnames) {
313  Plugin plugin = engine.getPlugin(pluginname);
314  if (plugin == null)
315  plugin = engine.getPlugin(pluginname + "Plugin");
316  if (plugin == null)
317  plugin = engine.getPlugin(pluginname + "Plugins");
318  if (plugin != null) {
319  plugins.add(plugin);
320  uses.add(plugin.getName());
321  }
322  }
323 
324  // unpack package plugins
325  // collect unpacked plugins in a separate set to prevent ConcurrentModificationException
326  // (occurs when adding elements to a HashSet while iterating over it)
327  Set<Plugin> unpackedPlugins = new HashSet<Plugin>();
328  for (Plugin plugin: plugins) {
329  if (plugin instanceof PackagePlugin) {
330  PackagePlugin pp = (PackagePlugin) plugin;
331  Set<String> set = pp.getEnclosedPluginNames();
332  for (String s: set) {
333  Plugin pl = engine.getPlugin(s);
334  if (pl != null)
335  unpackedPlugins.add(pl);
336  }
337  }
338  }
339  plugins.addAll(unpackedPlugins);
340 
341  // The ok-list collects all plugins with fulfilled dependencies
342  Set<Plugin> oklist = new HashSet<Plugin>();
343 
344  // get all plugins with no dependencies and add them to the ok-list
345  for (Plugin plugin: plugins) {
346  if (plugin.getDependencyNames().size() == 0)
347  oklist.add(plugin);
348  }
349 
350  // get all plugins whose plugins are already fulfilled
351  // loop this step until no more plugins are found
352  boolean flagChanged;
353  do {
354  flagChanged = false;
355  for (Plugin plugin : plugins) {
356  if (oklist.contains(plugin))
357  continue;
358  Set<String> depPl = plugin.getDependencyNames();
359  boolean foundAll = true;
360  for (String name : depPl) {
361  Plugin pl = engine.getPlugin(name);
362  if (pl == null || !oklist.contains(pl))
363  foundAll = false;
364  }
365  if (foundAll == true) {
366  oklist.add(plugin);
367  flagChanged = true;
368  }
369  }
370  } while (flagChanged == true);
371 
372  Set<String> okNameList = new HashSet<String>();
373  for (Plugin pl: oklist)
374  okNameList.add(pl.getName());
375 
376  return okNameList;
377  }
