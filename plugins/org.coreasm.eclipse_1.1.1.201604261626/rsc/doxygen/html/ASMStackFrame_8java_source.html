136  @Override
137  public IVariable[] getVariables() throws DebugException {
138  if (this.variables == null) {
139  Set<Location> updateLocations = new HashSet<Location>();
140  ArrayList<IVariable> variables = new ArrayList<IVariable>();
141  ArrayList<IVariable> backgrounds = new ArrayList<IVariable>();
142 
143  for (ASMUpdate update : state.getUpdates())
144  updateLocations.add(update.getLocation());
145 
146  variables.add(new ASMVariable(this, "Step", new ASMValue(this, "" + (getStep() < 0 ? -getStep() - 1 + "*" : getStep())), false));
147  variables.add(new ASMVariable(this, "Last Selected Agents", new ASMValue(this, getLastSelectedAgents().toString()), false));
148  variables.add(new ASMVariable(this, "Callstack", new ASMValue(this, state.getCallStack().toString()), false));
149 
150  for (Entry<String, Element> envVariable : state.getEnvVars().entrySet())
151  variables.add(new ASMVariable(this, envVariable.getKey(), new ASMValue(this, envVariable.getValue()), false));
152 
153  for (Entry<String, FunctionElement> function : state.getFunctions().entrySet()) {
154  String functionName = function.getKey();
155  FunctionElement functionElement = function.getValue();
156 
157  if (functionElement.isModifiable() && !AbstractStorage.FUNCTION_ELEMENT_FUNCTION_NAME.equals(functionName) && !AbstractStorage.RULE_ELEMENT_FUNCTION_NAME.equals(functionName)) {
158  for (Location location : functionElement.getLocations(functionName)) {
159  if (AbstractStorage.UNIVERSE_ELEMENT_FUNCTION_NAME.equals(functionName))
160  backgrounds.add(new ASMVariable(this, location.toString(), functionElement, new ASMValue(this, BooleanElement.TRUE), updateLocations.contains(location)));
161  else
162  variables.add(new ASMVariable(this, location.toString(), functionElement, new ASMValue(this, functionElement.getValue(location.args)), updateLocations.contains(location)));
163  }
164  }
165  }
166 
167  for (Entry<String, AbstractUniverse> universe : state.getUniverses().entrySet()) {
168  if (universe.getValue().isModifiable()) {
169  ArrayList<IVariable> universeVariables = new ArrayList<IVariable>();
170  String universeName = universe.getKey();
171  FunctionElement universeElement = universe.getValue();
172 
173  boolean containingValueChanged = false;
174  for (Location location : universeElement.getLocations(universeName)) {
175  if (updateLocations.contains(location))
176  containingValueChanged = true;
177  universeVariables.add(new ASMVariable(this, location.toString(), universeElement, new ASMValue(this, universeElement.getValue(location.args)), updateLocations.contains(location)));
178  }
179  variables.add(new ASMVariable(this, universeName, universeElement, new ASMValue(this, universeVariables.toArray(new IVariable[universeVariables.size()])), containingValueChanged));
180  }
181  }
182  variables.add(new ASMVariable(this, "Backgrounds", new ASMValue(this, backgrounds.toArray(new IVariable[backgrounds.size()])), false));
183  this.variables = variables.toArray(new IVariable[variables.size()]);
184  }
185  return this.variables;
186  }
